# Icon

The `DpIcon` component renders svg icons - mostly from fontawesome v5.

export const Template = (args, { argTypes }) => ({
    props: Object.keys(argTypes),
    components: { DpIcon },
    template: `<dp-icon v-bind="this.$props" />`
});
